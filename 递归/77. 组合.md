# 77. 组合
给定两个整数 n 和 k，返回范围 [1, n] 中所有可能的 k 个数的组合。
你可以按 任何顺序 返回答案。

示例 1：

    输入：n = 4, k = 2
    输出：
    [
      [2,4],
      [3,4],
      [2,3],
      [1,2],
      [1,3],
      [1,4],
    ]

示例 2：

    输入：n = 1, k = 1
    输出：[[1]]

提示：

 - 1 <= n <= 20
 - 1 <= k <= n
 
解答

    /**
     * @param {number} n
     * @param {number} k
     * @return {number[][]}
     */
    var combine = function(n, k) {
        let ans = []
    
        function dfs(cur, n, k, temp) {
            if (temp.length + (n - cur + 1) < k) {
                return;
            }
    
            if (temp.length === k) {
                ans.push(temp)
                return;
            }
    
            dfs(cur + 1, n, k, [...temp, cur])
    
            dfs(cur + 1, n, k, temp)
        }
        dfs(1, n, k, [])
        return ans;
    };
